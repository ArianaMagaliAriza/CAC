op1   db 2

      org 3000h
MUL:  MOV BX,[BX]
      MOV CH,BYTE PTR[BX]; OP2
      MOV BX,CX
      MOV BX,[BX]
      MOV AL, BYTE PTR[BX]; OP1
BUCLE:ADD CL,CH
      DEC AL
      JNZ BUCLE
      RET

      org 2000h

      MOV BX, OFFSET op2

      HLT
